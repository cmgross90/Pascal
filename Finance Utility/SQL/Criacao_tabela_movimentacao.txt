CREATE TABLE  MOVIMENTACAO
   (	CD_BANCO numeric(4,0), 
	CD_AGENCIA numeric(4,0), 
	NR_CONTA VARCHAR(15),
	IE_TIPO_MOVIMENTACAO VARCHAR(2),
	NR_MOVIMENTACAO NUMERIC(4),
	TIPO_CONTA varchar(15),
	
	NM_BENEFICIARIO VARCHAR(255),
	IE_CATEGORIA VARCHAR(2),
	IE_CATEGORIA_ESPECIFICA VARCHAR(2),
	DESCRICAO_DESPESA VARCHAR(255),
	NR_CHEQUE INTEGER,
	NR_DEPOSITO INTEGER,
	NR_TRANSFERENCIA INTEGER,
	DT_MOVIMENTACAO VARCHAR(10),
	VL_MONTANTE NUMERIC(7,2),
	NR_CONTA_TRANSF_DE VARCHAR(15),
	NR_CONTA_TRANSF_PARA VARCHAR(15),	 

	 CONSTRAINT MOVIMENTACAO_PK PRIMARY KEY (CD_BANCO, CD_AGENCIA, NR_CONTA, IE_TIPO_MOVIMENTACAO, NR_MOVIMENTACAO), 
	 CONSTRAINT MOVIMENTACAO_FK FOREIGN KEY (CD_BANCO) REFERENCES  BANCOS (CD_BANCO), 
	 CONSTRAINT MOVIMENTACAO_FK2 FOREIGN KEY (CD_AGENCIA) REFERENCES  AGENCIAS (CD_AGENCIA),
	 CONSTRAINT MOVIMENTACAO_FK3 FOREIGN KEY (TIPO_CONTA, NR_CONTA) REFERENCES  CONTAS (TIPO_CONTA, NR_CONTA)
   )
