CREATE TABLE  CONTA 
   (	CD_BANCO numeric(4,0), 
	CD_AGENCIA numeric(4,0), 
	TIPO_CONTA VARCHAR(15), 
	NR_CONTA VARCHAR(15), 
	NR_SALDO_INICIAL NUMeric(7,2), 
	 CONSTRAINT CONTA_PK PRIMARY KEY (TIPO_CONTA, NR_CONTA), 
	 CONSTRAINT CONTA_FK FOREIGN KEY (CD_BANCO)
	  REFERENCES  BANCOS (CD_BANCO), 
	 CONSTRAINT CONTA_FK2 FOREIGN KEY (CD_AGENCIA)
	  REFERENCES  AGENCIAS (CD_AGENCIA)
   )
